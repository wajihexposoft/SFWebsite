Animation Classes

All animations are gated by .js on <html> (added by the script block). Respects prefers-reduced-motion.

.reveal

Add to any block element. Fades up 20px on intersection. IntersectionObserver fires once, then unobserves.

<div class="reveal">...</div>

.stagger

Add to a grid/flex parent. Each direct child fades in with increasing delay (.04s increments, up to 6 children).

<div class="services-grid stagger">

.stat-num[data-count]

Adds animated counter. Set data-count="50" and data-suffix="+". Counts from 0 over 900ms easeOutCubic.

.code-row (line reveal)

Each .code-row reveals left-to-right with a 620ms initial delay + 44ms stagger per row. Index.html only.

Required Script Block (copy to every page)

(function () {
  document.documentElement.classList.add('js');
  var nav = document.querySelector('.nav');
  function syncNav() { nav.classList.toggle('nav-scrolled', window.scrollY > 60); }
  window.addEventListener('scroll', syncNav, { passive: true }); syncNav();
  var obs = new IntersectionObserver(function (entries) {
    entries.forEach(function (e) { if (e.isIntersecting) { e.target.classList.add('is-visible'); obs.unobserve(e.target); } });
  }, { threshold: 0.1 });
  document.querySelectorAll('.reveal, .stagger').forEach(function (el) { obs.observe(el); });
  function easeOutCubic(t) { return 1 - Math.pow(1 - t, 3); }
  document.querySelectorAll('.stat-num[data-count]').forEach(function (el) {
    var target = +el.dataset.count, suffix = el.dataset.suffix || '', dur = 900;
    var so = new IntersectionObserver(function (e) {
      if (e[0].isIntersecting) { so.disconnect(); var s = null;
        function tick(ts) { if (!s) s = ts; var p = Math.min((ts - s) / dur, 1); el.textContent = Math.round(easeOutCubic(p) * target) + suffix; if (p < 1) requestAnimationFrame(tick); }
        requestAnimationFrame(tick); }
    }, { threshold: 0.5 }); so.observe(el); });
})();
